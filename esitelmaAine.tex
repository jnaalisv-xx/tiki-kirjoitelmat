\subsection{Tilattoman ohjelmoinnin edut 1}
\begin{frame}
    \frametitle{Tilattoman ohjelmoinnin edut 1}
    \begin{block}{Testattaessa imperatiivista ohjelmaa}
    \begin{itemize}
     \item Asetetaan olion tila, ja mahdollinen ulkoinen tila jota olio käyttää
     \item Kutsutaan olion metodia
     \item Tutkitaan olion ja ohjelman tilaa     
     \item Mahdollisten testitapausten määrä = syötteet * tilat
   \end{itemize}
    \end{block}
    \begin{block}{Testattaessa puhdasta funktiota}
    \begin{itemize}
    \item Ei ole ulkoista eikä sisäistä tilaa mitä asettaa
    \item Testitapauksia on korkeintaan niin paljon kuin syötteitä
    \item Voidaan testata funktion palauttama arvo isolla syötejoukolla, \\
          imperatiivisessa ohjelmassa paluuarvon tulos voi riippua olion, \\
	  metodin tai ohjelman tilasta
    \end{itemize}	
    \end{block}
\end{frame}

\subsection{Tilattoman ohjelmoinnin edut 2}
\begin{frame}
\frametitle{Tilattoman ohjelmoinnin edut 2}
\begin{block}{Yhteenveto}
  \begin{itemize}
    \item Puhdas funktio on yksikkötestaajan "märkä uni"
    \item Ei väliä missä järjestyksessä funktioita kutsutaan
    \item Testien generointi voidaan automatisoida helpommin koska ei tarvitse asettaa tilaa
  \item Tilattomuus vähentää yllätysten määrää, koska funktiot eivät pääse vaikuttamaan ulkomaailmaansa.
    \item Funktionaalisessa ohjelmoinnissa ohjelma koostetaan pienistä helposti kattavasti testattavista funktioista. 
  \item Lyhyet helposti testattavat koodilohkot ovat nosteessa myös imperatiivisella puolella (TDD.)
   \item Virheenjäljitys huomattavasti helpompaa koska virheen ilmentyessä sen esiintyminen ei riipu ohjelman tilasta. Imperatiivisessa ohjelmassa virhe voi esiintyä vain ajoittain, kun ohjelma on tietyssä harvinaisessa tilassa.
  \end{itemize} 
    \end{block}
\end{frame}
